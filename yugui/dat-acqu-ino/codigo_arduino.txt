const int PINO_SENSOR_TEMPERATURA = A2;
const int PINO_SENSOR_LDR = A5;
float temperaturaCelsius;
int valorLuminosidade;

void setup() {
  Serial.begin(9600);
}

void loop() {
  float valorLeitura = analogRead(PINO_SENSOR_TEMPERATURA);
  float valorLuminosidade = analogRead(PINO_SENSOR_LDR);

  temperaturaCelsius = (valorLeitura * 5.0 / 1023.0) / 0.01;

  Serial.print(temperaturaCelsius);
  Serial.print(";");
  Serial.println(valorLuminosidade);

  delay(2000);
}
